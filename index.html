// 1. تحميل SheetJS ديناميكياً (CDN)
async function loadXLSX() {
  if (window.XLSX) return window.XLSX;
  return new Promise((resolve, reject) => {
    const s = document.createElement('script');
    s.src = 'https://cdnjs.cloudflare.com/ajax/libs/xlsx/0.18.5/xlsx.full.min.js';
    s.onload = () => resolve(window.XLSX);
    s.onerror = () => reject(new Error('XLSX load failed'));
    document.head.appendChild(s);
  });
}

// 2. دالة التصدير الرئيسية
async function exportITPToExcel(itpData, meta = {}) {
  const XLSX = await loadXLSX();

  // --- تعريف الأعمدة ---
  const COLUMNS = [
    { key: 'item',     header: 'Item No.',           width: 8  },
    { key: 'activity', header: 'Activity / الاختبار', width: 42 },
    { key: 'sub',      header: 'Description',         width: 30 },
    { key: 'qcs',      header: 'QCS 2024 Ref.',        width: 22 },
    { key: 'criteria', header: 'Acceptance Criteria',  width: 38 },
    { key: 'freq',     header: 'Frequency',            width: 20 },
    { key: 'lab',      header: 'Lab',                  width: 7  },
    { key: 'qc',       header: 'QC',                   width: 7  },
    { key: 'sc',       header: 'SC/CSC',               width: 9  },
    { key: 'type',     header: 'H/W/R',                width: 8  },
    { key: 'record',   header: 'Record',               width: 22 },
    { key: 'status',   header: 'Status',               width: 12 },
  ];

  // --- Header Rows ---
  const todayStr = new Date().toLocaleDateString('ar-QA');
  const headerMeta = [
    ['QatarSpec Pro — Inspection & Test Plan (ITP)',
      '', '', '', '', '', '', '', '', '', '', ''],
    ['المشروع:', meta.project || '—', '', 'رقم ITP:',
      meta.itpNum || 'ITP-001', '', 'التاريخ:', todayStr, '', '', '', ''],
    ['المهندس:', meta.engineer || '—', '', 'المرجع:',
      'QCS 2024', '', 'الإصدار:', '1.0', '', '', '', ''],
    ['', '', '', '', '', '', '', '', '', '', '', ''],
    COLUMNS.map(c => c.header),
  ];

  // --- Data Rows: Normalized ---
  const dataRows = itpData.map(r => COLUMNS.map(c => {
    if (c.key === 'criteria') return (r.crit || r[c.key] || '').replace(/\n/g, ' | ');
    if (c.key === 'type')
      return r.type === 'H' ? 'Hold Point' : r.type === 'W' ? 'Witness Point' : 'Review Point';
    if (c.key === 'record') return 'ITP Register / Lab Report';
    if (c.key === 'status') return 'Pending';
    return r[c.key] || r[Object.keys(r).find(k => k === c.key)] || '';
  }));

  // --- Sheet 1: ITP Main ---
  const ws1 = XLSX.utils.aoa_to_sheet([...headerMeta, ...dataRows]);
  ws1['!cols'] = COLUMNS.map(c => ({ wch: c.width }));
  ws1['!merges'] = [
    { s:{r:0,c:0}, e:{r:0,c:11} },  // Title row full width
    { s:{r:1,c:0}, e:{r:1,c:1} },   // Project name
    { s:{r:2,c:0}, e:{r:2,c:1} },   // Engineer
  ];

  // --- Style Header Row (Row 5) ---
  const headerRowIdx = 4; // 0-indexed
  COLUMNS.forEach((_, ci) => {
    const cellAddr = XLSX.utils.encode_cell({ r: headerRowIdx, c: ci });
    if (!ws1[cellAddr]) return;
    ws1[cellAddr].s = {
      fill: { fgColor: { rgb: '7A1515' } },
      font: { color: { rgb: 'C9A84C' }, bold: true, sz: 10 },
      alignment: { horizontal: 'right', wrapText: true },
    };
  });

  // --- Sheet 2: Hold Points Only ---
  const holdRows = itpData.filter(r => r.type === 'H' || r.type === 'W');
  const ws2 = XLSX.utils.aoa_to_sheet([
    ['Hold Points & Witness Points — Summary'],
    ['Item', 'Activity', 'QCS Ref.', 'Acceptance Criteria', 'Type', 'Record'],
    ...holdRows.map(r => [
      r.item, r.activity || r.act || '',
      r.qcs,
      (r.crit || '').replace(/\n/g, ' | '),
      r.type === 'H' ? 'Hold Point' : 'Witness Point',
      'ITP Register'
    ])
  ]);
  ws2['!cols'] = [{wch:6},{wch:40},{wch:22},{wch:38},{wch:14},{wch:22}];

  // --- Sheet 3: Test Frequency Schedule ---
  const ws3 = XLSX.utils.aoa_to_sheet([
    ['Test Frequency Schedule — جدول الاختبارات'],
    ['م', 'الاختبار', 'المعيار', 'التكرار', 'الوحدة', 'QCS Clause', 'النوع'],
    ...itpData.map(r => [
      r.item, r.activity || r.act || '',
      r.qcs, r.freq || '',
      'm² / tonne / LM',
      r.qcs,
      r.type === 'H' ? 'Hold' : r.type === 'W' ? 'Witness' : 'Review'
    ])
  ]);
  ws3['!cols'] = [{wch:6},{wch:40},{wch:22},{wch:20},{wch:14},{wch:22},{wch:10}];

  // --- Build Workbook & Export ---
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws1, 'ITP Main');
  XLSX.utils.book_append_sheet(wb, ws2, 'Hold Points');
  XLSX.utils.book_append_sheet(wb, ws3, 'Test Schedule');

  const filename = `ITP-${meta.itpNum || 'export'}-${Date.now()}.xlsx`;
  XLSX.writeFile(wb, filename);
  return filename;
}

async function exportNCRLog(ncrList, meta = {}) {
  const XLSX = await loadXLSX();
  const NCR_COLS = [
    {key:'num',    header:'NCR No.',        width:10},
    {key:'date',   header:'Date',           width:12},
    {key:'loc',    header:'Location + Grid',width:28},
    {key:'clause', header:'QCS Clause',     width:20},
    {key:'desc',   header:'Non-Conformance',width:40},
    {key:'root',   header:'Root Cause',     width:22},
    {key:'correct',header:'Corrective Action',width:35},
    {key:'prevent',header:'Preventive Action',width:35},
    {key:'resp',   header:'Responsible',    width:18},
    {key:'close',  header:'Target Close',   width:14},
    {key:'status', header:'Status',         width:12},
  ];
  const ws = XLSX.utils.aoa_to_sheet([
    ['NCR Log — Non-Conformance Register', ...Array(10).fill('')],
    ['المشروع:', meta.project||'—', '', 'المرجع:', 'QCS 2024 / ISO 9001 Cl.10.2',
      ...Array(6).fill('')],
    ['','','','','','','','','','',''],
    NCR_COLS.map(c => c.header),
    ...ncrList.map(r => NCR_COLS.map(c => r[c.key] || ''))
  ]);
  ws['!cols'] = NCR_COLS.map(c => ({wch: c.width}));
  ws['!merges'] = [{s:{r:0,c:0},e:{r:0,c:10}}];
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws, 'NCR Log');
  XLSX.writeFile(wb, `NCR-Log-${meta.project||'Project'}-${Date.now()}.xlsx`);
}
